\section*{AceWiki Grammar}

\noindent - Tobias Kuhn, 2 August 2010 - \vspace{2mm}

\noindent Below, the grammar rules of the AceWiki grammar are shown: \vspace{2mm}

\subsection*{Texts and Sentences}

\noindent 'text' stands for a complete text consisting of an arbitrary number of complete
		sentences (including zero): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{text}
}{
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{text}
}{
  \scat{complete\_sentence}
  \scat{text}
}$
\vspace{2mm}

}
\noindent A complete sentence is represented by the category 'complete\_sentence' and is either
		a declarative sentence that ends with a full stop or a question ending with a question mark: \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{complete\_sentence}
}{
  \scat{sentence}
  \term{.}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{complete\_sentence}
}{
  \scopeopensymb{}
  \cat{simple\_sentence\_2}{\featc{whin}{--}\featc{whout}{+}}
  \term{?}
}$
\vspace{2mm}

}
\noindent General sentences are represented by 'sentence': \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence}
}{
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence}
}{
  \scopeopensymb{}
  \term{for every}
  \cat{nc}{\featc{subj}{--}}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence}
}{
  \scopeopensymb{}
  \term{if}
  \scat{sentence\_coord\_1}
  \term{then}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
\noindent Sentences can be coordinated using "or" ('sentence\_coord\_1') and "and"
		('sentence\_coord\_2'): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_1}
}{
  \scat{sentence\_coord\_2}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence\_coord\_1}
}{
  \scopeopensymb{}
  \scat{sentence\_coord\_2}
  \term{or}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_2}
}{
  \scat{simple\_sentence\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_2}
}{
  \scat{simple\_sentence\_1}
  \term{and}
  \scat{sentence\_coord\_2}
}$
\vspace{2mm}

}
\noindent Uncoordinated sentences are represented in two levels by 'simple\_sentence\_1' and
		'simple\_sentence\_2': \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{simple\_sentence\_1}
}{
  \scopeopensymb{}
  \term{it is false that}
  \cat{simple\_sentence\_2}{\featc{whin}{--}\featc{whout}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there is}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{--}\featc{subj}{--}\featc{whin}{--}\featc{whout}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there is}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{--}\featc{subj}{--}\featc{whin}{--}\featc{whout}{--}}
  \term{such that}
  \scat{simple\_sentence\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there are}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{+}\featc{subj}{--}\featc{whin}{--}\featc{whout}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \cat{simple\_sentence\_2}{\featc{whin}{--}\featc{whout}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{simple\_sentence\_2}{\featv{whin}{1}\featv{whout}{2}}
}{
  \cat{np}{\featc{case}{nom}\featv{id}{3}\featv{pl}{4}\featc{subj}{--}\featv{whin}{1}\featv{whout}{5}}
  \cat{vp\_coord\_1}{\featv{pl}{4}\featv{subj}{3}\featv{whin}{5}\featv{whout}{2}}
}$
\vspace{2mm}

}
\subsection*{Verb Phrases}

\noindent Like sentences, verb phrases can be coordinated using "or" ('vp\_coord\_1') and "and"
		('vp\_coord\_2'): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_1}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}
}{
  \cat{vp\_coord\_2}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{vp\_coord\_1}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}
}{
  \scopeopensymb{}
  \cat{vp\_coord\_2}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{5}}
  \term{or}
  \cat{vp\_coord\_1}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{5}\featv{whout}{4}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_2}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}
}{
  \cat{vp}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_2}{\featv{pl}{2}\featv{subj}{1}\featv{whin}{3}\featv{whout}{4}}

\vspace{2mm}

}

{\scriptsize
\noindent$

}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{vp}{\featc{exist}{+}\featv{pl}{3}\featv{rel}{2}\featv{subj}{1}\featv{whin}{4}\featv{whout}{5}}
}{
  \cat{v}{\featc{be}{--}\featc{exist}{+}\featv{pl}{3}\featv{rel}{2}\featv{subj}{1}\featc{vform}{fin}\featv{whin}{4}\featv{whout}{5}}
}$
\vspace{2mm}

}

		complementing noun phrase or adjective complement: \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{--}\featc{copula}{--}\featv{exist}{1}\featv{pl}{2}\featv{vform}{3}\featv{whin}{4}\featv{whout}{4}}
}{
  \cat{verb}{\featc{be}{--}\featv{exist}{1}\featv{pl}{2}\featc{vcat}{itr}\featv{vform}{3}}

\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{--}\featc{copula}{--}\featv{embv}{6}\featv{exist}{2}\featv{pl}{4}\featv{rel}{3}\featv{subj}{1}\featv{vform}{5}\featv{whin}{7}\featv{whout}{8}}
}{
  \cat{verb}{\featc{be}{--}\featv{exist}{2}\featv{pl}{4}\featc{vcat}{tr}\featv{vform}{5}}
  \cat{np}{\featc{case}{acc}\featv{embv}{6}\featv{rel}{3}\featv{subj}{1}\featc{vcat}{tr}\featv{whin}{7}\featv{whout}{8}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{+}\featc{copula}{--}\featv{embv}{3}\featv{rel}{2}\featv{subj}{1}\featv{whin}{4}\featv{whout}{5}}
}{

  \cat{np}{\featc{case}{acc}\featc{copula}{--}\featv{embv}{3}\featv{rel}{2}\featv{subj}{1}\featv{whin}{4}\featv{whout}{5}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{+}\featc{copula}{+}\featv{embv}{3}\featv{rel}{2}\featv{subj}{1}\featv{whin}{4}\featv{whout}{5}}
}{

\noindent$

\vspace{2mm}

}

\noindent$
\ruleid
\nrule{
