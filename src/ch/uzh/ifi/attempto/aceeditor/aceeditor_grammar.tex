\section*{ACE Editor Grammar}

\noindent - Tobias Kuhn, 26 November 2010 - \vspace{2mm}

\noindent Below, the grammar rules of the ACE Editor grammar are shown: \vspace{2mm}

\subsection*{Texts and Sentences}

\noindent 'text' stands for a complete text consisting of an arbitrary number of complete
		sentences (including zero): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{text}
}{
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{text}
}{
  \scat{complete\_sentence}
  \scat{text}
}$
\vspace{2mm}

}
\noindent A complete sentence is represented by the category 'complete\_sentence' and is either
		a declarative sentence that ends with a full stop or a question ending with a question mark: \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{complete\_sentence}
}{
  \scat{sentence}
  \term{.}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{complete\_sentence}
}{
  \scopeopener{}
  \cat{simple\_sentence\_2}{\featc{qu}{+}\featc{whin}{--}\featc{whout}{+}}
  \term{?}
}$
\vspace{2mm}

}
\noindent General sentences are represented by 'sentence': \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence}
}{
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence}
}{
  \scopeopener{}
  \term{for every}
  \cat{nc}{\featc{qu}{--}\featc{subj}{--}}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence}
}{
  \scopeopener{}
  \term{if}
  \scat{sentence\_coord\_1}
  \term{then}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
\noindent Sentences can be coordinated using "or" ('sentence\_coord\_1') and "and"
		('sentence\_coord\_2'): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_1}
}{
  \scat{sentence\_coord\_2}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{sentence\_coord\_1}
}{
  \scopeopener{}
  \scat{sentence\_coord\_2}
  \term{or}
  \scat{sentence\_coord\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_2}
}{
  \scat{simple\_sentence\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{sentence\_coord\_2}
}{
  \scat{simple\_sentence\_1}
  \term{and}
  \scat{sentence\_coord\_2}
}$
\vspace{2mm}

}
\noindent Uncoordinated sentences are represented in two levels by 'simple\_sentence\_1' and
		'simple\_sentence\_2': \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\scrule{
  \scat{simple\_sentence\_1}
}{
  \scopeopener{}
  \term{it is false that}
  \cat{simple\_sentence\_2}{\featc{qu}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there is}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{--}\featc{qu}{--}\featc{subj}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there is}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{--}\featc{qu}{--}\featc{subj}{--}}
  \term{such that}
  \scat{simple\_sentence\_1}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \term{there are}
  \cat{np}{\featc{case}{nom}\featc{def}{--}\featc{exist}{+}\featc{pl}{+}\featc{qu}{--}\featc{subj}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \scat{simple\_sentence\_1}
}{
  \cat{simple\_sentence\_2}{\featc{qu}{--}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{simple\_sentence\_2}{\featv{qu}{1}\featv{whin}{2}\featv{whout}{3}}
}{
  \cat{np}{\featc{case}{nom}\featv{id}{4}\featv{pl}{5}\featv{qu}{1}\featc{subj}{--}\featv{whin}{2}\featv{whout}{6}}
  \cat{vp\_coord\_1}{\featv{pl}{5}\featv{qu}{1}\featv{subj}{4}\featv{whin}{6}\featv{whout}{3}}
}$
\vspace{2mm}

}
\subsection*{Verb Phrases}

\noindent Like sentences, verb phrases can be coordinated using "or" ('vp\_coord\_1') and "and"
		('vp\_coord\_2'): \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_1}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}{
  \cat{vp\_coord\_2}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{vp\_coord\_1}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}{
  \scopeopener{}
  \cat{vp\_coord\_2}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{6}}
  \term{or}
  \cat{vp\_coord\_1}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{6}\featv{whout}{5}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_2}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}{
  \cat{vp}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{vp\_coord\_2}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{5}}
}{
  \cat{vp}{\featv{pl}{1}\featv{qu}{2}\featv{subj}{3}\featv{whin}{4}\featv{whout}{6}}

}$
\vspace{2mm}

}
\noindent Uncoordinated verb phrases represented by 'vp' can use an auxiliary verb and can have
		verb phrase modifiers: \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{vp}{\featv{exist}{1}\featv{pl}{2}\featv{qu}{3}\featv{rel}{4}\featv{subj}{5}\featv{whin}{6}\featv{whout}{7}}

  \cat{aux}{\featv{be}{8}\featv{exist}{1}\featv{pl}{2}}

  \cat{vmod}{\featv{copula}{9}\featv{embv}{10}\featv{qu}{3}\featv{subj}{5}\featv{whin}{11}\featv{whout}{7}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\scrule{
  \cat{vp}{\featc{exist}{+}\featv{pl}{1}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}
}{
  \cat{v}{\featc{be}{--}\featv{copula}{7}\featv{embv}{8}\featc{exist}{+}\featv{pl}{1}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featc{vform}{fin}\featv{whin}{5}\featv{whout}{9}}
  \cat{vmod}{\featv{copula}{7}\featv{embv}{8}\featv{qu}{2}\featv{subj}{4}\featv{whin}{9}\featv{whout}{6}}
}$
\vspace{2mm}

}
\noindent The category 'v' represents the main verb or - if "be" is used as a copula verb - the
		complementing noun phrase or adjective complement: \vspace{2mm}

{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{--}\featc{copula}{--}\featv{pl}{1}\featv{vform}{2}\featv{whin}{3}\featv{whout}{3}}
}{
  \cat{verb}{\featc{be}{--}\featv{pl}{1}\featc{vcat}{itr}\featv{vform}{2}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{--}\featc{copula}{--}\featv{embv}{1}\featv{pl}{2}\featv{qu}{3}\featv{rel}{4}\featv{subj}{5}\featv{vform}{6}\featv{whin}{7}\featv{whout}{8}}
}{
  \cat{verb}{\featc{be}{--}\featv{pl}{2}\featc{vcat}{tr}\featv{vform}{6}}

\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{+}\featc{copula}{--}\featv{embv}{1}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}
}{
  \cat{verb}{\featc{be}{+}\featc{vcat}{tr}}
  \term{by}
  \cat{np}{\featc{case}{acc}\featc{copula}{--}\featv{embv}{1}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{+}\featc{copula}{+}\featv{embv}{1}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}
}{
  \cat{np}{\featc{case}{acc}\featc{copula}{+}\featv{embv}{1}\featc{of}{+}\featc{pl}{--}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}
}$
\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{
  \cat{v}{\featc{be}{+}\featc{copula}{+}\featv{embv}{1}\featc{pl}{--}\featv{qu}{2}\featv{rel}{3}\featv{subj}{4}\featv{whin}{5}\featv{whout}{6}}

\vspace{2mm}

}
{\scriptsize
\noindent$
\ruleid
\nrule{

}{
  \scat{adj\_coord}

\vspace{2mm}

}

\noindent$
